Цикл while
Выполняется до тех пор, пока логическое выражение равно True
while 1 == 1: # условие всегда истинно, цикл остановится не скоро...
 print("1 == 1")

Цикл for
Перебирает все объекты в диапазоне
for i in range(0, 10 + 1):
 print(i) # цикл выведет числа от 0 до 10 for i in range(100, 0, -1):
 print(i) # цикл выведет числа от 100 до 1

Операторы циклов

break позволяет прервать цикл
for i in range(1, 100 + 1):
 if i == 50:
  break # когда i достигнет 50, цикл остановится
 print(i) i = 1
while i < 100 + 1:
 if i == 50:
  break # когда i достигнет 50, цикл остановится
 print(i)
 i += 1
continue позволяет перейти к следующей итерации
for i in range(1, 100 + 1):
 if i == 50:
  continue
 print(i) # выведет все числа от 1 до 100 кроме 50 i = 1
while i < 100 + 1:
 if i == 50:
  continue
 print(i)
# выведет все числа от 1 до 100 кроме 50
 i += 1
