+ : сложение
a = 1 + 1 # a = 2

- : вычитание
a = 156 - 13 # a = 143

* : умножение
s = 12 * 14 # s = 168

/ : деление
r = 123 / 3 # r = 41.0

** : возведение в степень
k = 2 ** 4 # k = 16

// : целая часть от деления
v = 15 // 4 # v = 3

% : Остаток от деления
q = 7 % 4 # q = 3

# - начало комментария
print() : операция вывода
input() : операция ввода (возвращает строку), int(input()) для целых чисел
