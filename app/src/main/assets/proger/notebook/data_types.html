int : тип данных, обозначающий целые числа
a = 123

float : тип данных, означающий числа с плавающей точкой
b = 3.1415

string : тип данных, означающий строки
name = "evjeny"

boolean : тип данных, означающий логические переменные (может принимать только два значения)
left = True
right = False

Преобразование типов
a = int("155") # преобразование из string в int
a = float(input()) # преобразование из string в float
a = str(13) # преобразование из int в string
a = bool(0.123) # преобразование из float в boolean
